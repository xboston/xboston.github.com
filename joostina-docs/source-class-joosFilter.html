46:      *
47:      * @param mixed $mixed        строка массив или объект для обезопасивания
48:      * @param const $quote_style  тип зашиты - ENT_COMPAT, ENT_QUOTES или ENT_NOQUOTES

52:      */
53:     public static function make_safe(&$mixed, $quote_style = ENT_QUOTES, $exclude_keys = '') {
54:         if (is_object($mixed)) {
55:             foreach (get_object_vars($mixed) as $k => $v) {
56:                 if (is_array($v) || is_object($v) || $v == null || substr($k, 1, 1) == '_') {
57:                     continue;
58:                 }
59:                 if (is_string($exclude_keys) && $k == $exclude_keys) {
60:                     continue;
61:                 } else if (is_array($exclude_keys) && in_array($k, $exclude_keys)) {
62:                     continue;
63:                 }
64:                 $mixed->$k = htmlspecialchars($v, $quote_style, 'UTF-8');
65:             }
66:         } elseif (is_string($mixed)) {
67:             return htmlspecialchars($mixed, $quote_style, 'UTF-8');
68:         }
69:     }
