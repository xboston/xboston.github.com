222:     /**
223:      * Returns the compiled regular expression for the route. This translates
224:      * keys and optional groups to a proper PCRE regular expression.
225:      *
226:      *     $compiled = self::compile(
227:      *        '<controller>(/<action>(/<id>))',
228:      *         array(
229:      *           'controller' => '[a-z]+',
230:      *           'id' => '\d+',
231:      *         )
232:      *     );
233:      *
234:      * @return  string
235:      * @uses    self::REGEX_ESCAPE
236:      * @uses    self::REGEX_SEGMENT
237:      */
238:     private static function compile($uri, array $regex = NULL) {
239:         if (!is_string($uri)) {
240:             return;
241:         }
242: 
243:         // The URI should be considered literal except for keys and optional parts
244:         // Escape everything preg_quote would escape except for : ( ) < >
245:         $expression = preg_replace('#' . self::REGEX_ESCAPE . '#', '\\\\$0', $uri);
246: 
247:         if (strpos($expression, '(') !== FALSE) {
248:             // Make optional parts of the URI non-capturing and optional
249:             $expression = str_replace(array('(', ')'), array('(?:', ')?'), $expression);
250:         }
251: 
252:         // Insert default regex for keys
253:         $expression = str_replace(array('<', '>'), array('(?P<', '>' . self::REGEX_SEGMENT . ')'), $expression);
254: 
255:         // правила краткой записи регулярок роутинга
256:         $rules = array(
257:             ':any' => '.+?',
258:             ':maybe' => '.*?',
259:             ':digit' => '[\d]+',
260:             ':alpha' => '[a-zA-Z]+',
261:             ':rus_alpha' => '[a-zA-Zа-яА-ЯёЁ]+',
262:             ':word' => '[\w-_]+',
263:         );
264: 
265:         if ($regex) {
266:             $search = $replace = array();
267:             foreach ($regex as $key => $value) {
268: 
269:                 $value = strtr($value, $rules);
270: 
271:                 $search[] = "<$key>" . self::REGEX_SEGMENT;
272:                 $replace[] = "<$key>$value";
273:             }
274: 
275:             // Replace the default regex with the user-specified regex
276:             $expression = str_replace($search, $replace, $expression);
277:         }
278: 
279:         return '#^' . $expression . '$#uD';
280:     }

355:     /**
356:      * Tests if the route matches a given URI. A successful match will return
357:      * all of the routed parameters as an array. A failed match will return
358:      * boolean FALSE.
359:      *
360:      *     // Params: controller = users, action = edit, id = 10
361:      *     $params = $route->matches('users/edit/10');
362:      *
363:      * This method should almost always be used within an if/else block:
364:      *
365:      *     if ($params = $route->matches($uri))
366:      *     {
367:      *         // Parse the parameters
368:      *     }
369:      *
370:      * @param   string  URI to match
371:      *
372:      * @return  array   on success
373:      * @return  FALSE   on failure
374:      */
375:     protected function matches($uri) {
376: 
377:         if (!preg_match($this->_route_regex, $uri, $matches)) {
378:             return FALSE;
379:         }
380: 
381:         $params = array();
382:         foreach ($matches as $key => $value) {
383:             if (is_int($key)) {
384:                 // Skip all unnamed keys
385:                 continue;
386:             }
387: 
388:             // Set the value for all matched keys
389:             $params[$key] = $value;
390:         }
391: 
392: 
393:         foreach ($this->_defaults as $key => $value) {
394:             if (!isset($params[$key]) OR $params[$key] === '') {
395:                 // Set default values for any key that was not matched
396:                 $params[$key] = $value;
397:             }
398:         }
399: 
400:         return $params;
401:     }

403:     /**
404:      * Generates a URI for the current route based on the parameters given.
405:      *
406:      *     // Using the "default" route: "users/profile/10"
407:      *     $route->uri(array(
408:      *         'controller' => 'users',
409:      *         'action'     => 'profile',
410:      *         'id'         => '10'
411:      *     ));
412:      *
413:      * @param   array   URI parameters
414:      *
415:      * @return  string
416:      * @throws  joosException
417:      * @uses    self::REGEX_Key
418:      */
419:     protected function uri(array $params = NULL) {
420:         // Start with the routed URI
421:         $uri = $this->_uri;
422: 
423:         // если в ссылке нет динамических параметров - сразу её возвратим
424:         if (strpos($uri, '<') === FALSE AND strpos($uri, '(') === FALSE) {
425:             return $uri;
426:         }
427: 
428:         while (preg_match('#' . self::REGEX_KEY . '#', $uri, $match)) {
429:             list( $key, $param ) = $match;
430:             if (!isset($params[$param])) {
431:                 // Look for a default
432:                 if (isset($this->_defaults[$param])) {
433:                     $params[$param] = $this->_defaults[$param];
434:                 } else {
435:                     // отсутствуют требуемые параметры
436:                     throw new joosException('Требуемый параметр :param не найден в полученных данных для условия :uri',

438:                                 ':param' => $param,
439:                                 ':uri' => joosFilter::htmlspecialchars($this->_uri)
440:                             )
441:                     );
442:                 }
443:             }
444: 
445:             $uri = str_replace($key, $params[$param], $uri);
446:         }
447: 
448:         // чистка от лишних и дублирующихся /
449:         $uri = preg_replace('#//+#', '/', rtrim($uri, '/'));
450: 
451:         return $uri;
452:     }
