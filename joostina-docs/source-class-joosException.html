 21: 
 22:     const CONTEXT_RADIUS = 10;
 23: 
 24:     public function __construct($message = '', array $params = array()) {
 25:         parent::__construct(strtr($message, $params));
 26: 
 27:         if (isset($params[':error_file'])) {
 28:             $this->file = $params[':error_file'];
 29:         }
 30: 
 31:         if (isset($params[':error_line'])) {
 32:             $this->line = $params[':error_line'];
 33:         }
 34: 
 35:         if (isset($params[':error_code'])) {
 36:             $this->code = $params[':error_code'];
 37:         }
 38: 
 39:         $this->__toString();
 40:     }
 41: 
 42:     private function get_file_context() {
 43: 
 44:         $file = $this->getFile();
 45:         $line_number = $this->getLine();
 46: 
 47:         $context = array();
 48:         $i = 0;
 49:         foreach (file($file) as $line) {
 50:             $i++;
 51:             if ($i >= $line_number - self::CONTEXT_RADIUS && $i <= $line_number + self::CONTEXT_RADIUS) {
 52:                 if ($i == $line_number) {
 53:                     $context[] = ' >>   ' . $i . "\t" . $line;
 54:                 } else {
 55:                     $context[] = "\t" . $i . "\t" . $line;
 56:                 }
 57:             }
 58:             if ($i > $line_number + self::CONTEXT_RADIUS) {
 59:                 break;
 60:             }
 61:         }
 62:         return "\n" . implode("", $context);
 63:     }
 64: 
 65:     public function __toString() {
 66:         // очистим всю вышестоящую буферизацию без вывода её в браузер
 67:         !ob_get_level() ? : ob_end_clean();
 68: 
 69:         parent::__toString();
 70:         echo joosRequest::is_ajax() ? $this->to_json() : $this->show();
 71:         die();
 72:     }
 73: 
 74:     private function show() {
 75:         return JDEBUG ? $this->create() : $this->render();
 76:     }
 77: 
 78:     private function create() {
 79: 
 80:         if (headers_sent()) {
 81:             !ob_get_level() ? : ob_end_clean();
 82:         } else {
 83:             header('Content-type: text/html; charset=UTF-8');
 84:         }
 85: 
 86:         $message = nl2br($this->getMessage());
 87:         $result = <<<HTML
 88:   <style>
 89:     body { background-color: #fff; color: #333; }
 90:     body, p, ol, ul, td { font-family: verdana, arial, helvetica, sans-serif; font-size: 13px; line-height: 25px; }
 91:     pre { background-color: #eee; padding: 10px; font-size: 11px; line-height: 18px; }
 92:     a { color: #000; }
 93:     a:visited { color: #666; }
 94:     a:hover { color: #fff; background-color:#000; }
 95:   </style>
 96: <div style="width:99%; position:relative">
 97: <h2 id='Title'>{$message}</h2>
 98: <div id="Context" style="display: block;"><h3>Ошибка с кодом {$this->getCode()} в файле '{$this->getFile()}' в строке {$this->getLine()}:</h3><pre>{$this->prepare($this->get_file_context())}</pre></div>
 99: <div id="Trace"><h2>Стэк вызовов</h2><pre>{$this->getTraceAsString()}</pre></div>
100: HTML;
101:         $result .= "</div></div>";
102:         return $result;
103:     }
104: 
105:     protected function prepare($content) {
106:         return joosFilter::htmlspecialchars($content);
107:     }
108: 
109:     /**
110:      * Возврат информации об ошибки в JSON-сериализованном виде
111:      *
112:      * @return json string строка с кодом ошибки закодированная в JSON
113:      */
114:     private function to_json() {
115:         $response = array(
116:             'code' => ( $this->getCode() != 0 ) ? $this->getCode() : 500,
117:             'message' => $this->getMessage()
118:         );
119: 
120:         return joosText::json_encode($response);
121:     }

123:     private function render() {
124:         $message = $this->message;
125: 
126:         $file = (!JDEBUG && $this->code != 500) ? sprintf('%s/app/templates/system/500.php', JPATH_BASE) : sprintf('%s/app/templates/system/exception.php', JPATH_BASE);
127: 
128:         require $file;
129:     }

136:                     ':error_file' => $file,
