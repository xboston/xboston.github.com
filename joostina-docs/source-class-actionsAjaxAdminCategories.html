 57:     public static function slug_generator() {
 58: 
 59:         $cats = new modelCategories();
 60: 
 61:         //$cat_id = joosRequest::request('cat_id', 0);
 62:         $cat_name  = joosRequest::request( 'cat_name' , '' );
 63:         $parent_id = joosRequest::request( 'parent_id' , 0 );
 64: 
 65:         if ( !$cat_name ) {
 66:             echo json_encode( array ( 'error' => 'Введите имя категории' ) );
 67:             return;
 68:         }
 69: 
 70:         //Если  в качестве родителя выбран корень
 71:         if ( $parent_id == 1 ) {
 72:             //просто транслитерируем название категории
 73:             $slug = joosText::text_to_url( $cat_name );
 74:         }
 75: 
 76:             //иначе - построим путь c учётом выбранного родителя
 77:         else {
 78: 
 79:             $segments = array ();
 80: 
 81:             $path     = $cats->get_path_from_root( $parent_id );
 82:             unset( $path[1] );
 83: 
 84:             $_repeat = '';
 85:             foreach ( $path as $_cat ) {
 86:                 $_slug      = $_cat['slug'] ? $_cat['slug'] . '/' : joosText::text_to_url( $_cat['name'] . '/' );
 87:                 $_slug      = str_replace( $_repeat , '' , $_slug );
 88:                 $segments[] = $_slug;
 89:                 $_repeat .= $_slug;
 90:             }
 91: 
 92:             $segments[] = joosText::text_to_url( $cat_name );
 93:             $slug       = implode( '' , $segments );
 94:         }
 95: 
 96:         echo json_encode( array ( 'slug' => $slug ) );
 97:         return;
 98:     }
