 59:     public static function slug_generator() {
 60: 
 61:         $cats = new modelCategories();
 62: 
 63:         //$cat_id = joosRequest::request('cat_id', 0);
 64:         $cat_name  = joosRequest::request( 'cat_name' , '' );
 65:         $parent_id = joosRequest::request( 'parent_id' , 0 );
 66: 
 67:         if ( !$cat_name ) {
 68:             echo json_encode( array ( 'error' => 'Введите имя категории' ) );
 69:             return;
 70:         }
 71: 
 72:         //Если  в качестве родителя выбран корень
 73:         if ( $parent_id == 1 ) {
 74:             //просто транслитерируем название категории
 75:             $slug = joosText::str_to_url( $cat_name );
 76:         }
 77: 
 78:             //иначе - построим путь c учётом выбранного родителя
 79:         else {
 80: 
 81:             $segments = array ();
 82: 
 83:             $path     = $cats->get_path_from_root( $parent_id );
 84:             unset( $path[1] );
 85: 
 86:             $_repeat = '';
 87:             foreach ( $path as $_cat ) {
 88:                 $_slug      = $_cat['slug'] ? $_cat['slug'] . '/' : joosText::str_to_url( $_cat['name'] . '/' );
 89:                 $_slug      = str_replace( $_repeat , '' , $_slug );
 90:                 $segments[] = $_slug;
 91:                 $_repeat .= $_slug;
 92:             }
 93: 
 94:             $segments[] = joosText::str_to_url( $cat_name );
 95:             $slug       = implode( '' , $segments );
 96:         }
 97: 
 98:         echo json_encode( array ( 'slug' => $slug ) );
 99:         return;
100:     }
