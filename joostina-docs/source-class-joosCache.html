 30:     public function __construct($type = false, $data_store = false) {
 31: 
 32:         $type = $type ? $type : joosConfig::get2('cache', 'handler');
 33: 
 34:         switch ($type) {
 35:             case 'file':
 36: 
 37:                 $data_store = $data_store ? $data_store : joosConfig::get2('cache', 'cachepath');
 38: 
 39:                 $exists = file_exists($data_store);
 40:                 if (!$exists && !is_writable(dirname($data_store))) {
 41:                     throw new joosCacheException(sprintf(__('Каталог кеширования %s недоступен для записи'), $data_store));
 42:                 }
 43:                 if ($exists && !is_writable($data_store)) {
 44:                     throw new joosCacheException(sprintf(__('Файл кеша %s недоступен для записи'), $data_store));
 45:                 }
 46:                 $this->data_store = $data_store;
 47:                 if ($exists) {
 48:                     $this->cache = unserialize(file_get_contents($data_store));
 49:                 } else {
 50:                     $this->cache = array();
 51:                 }
 52:                 $this->state = 'clean';
 53:                 break;
 54: 
 55:             case 'apc':
 56:             case 'xcache':
 57:             case 'memcache':
 58:                 if (!extension_loaded($type)) {
 59:                     throw new joosCacheException(sprintf(__('Расширение кеширования %s недоступно либо не установлено'), $type));
 60:                 }
 61:                 if ($type == 'memcache') {
 62: 
 63:                     if ($data_store == false) {
 64:                         $data_store = new Memcache();
 65:                         $data_store->connect(joosConfig::get2('cache', 'memcache_host'), joosConfig::get2('cache', 'memcache_port'));
 66:                     }
 67: 
 68:                     if (!$data_store instanceof Memcache) {
 69:                         throw new joosCacheException(__('Объект кеширования не является допустимым объектом Memcache'));
 70:                     }
 71:                     $this->data_store = $data_store;
 72:                 }
 73:                 break;
 74: 
 75:             default:
 76:                 throw new joosCacheException(sprintf(__('Кеширующая система не поддерживает %s, разрешено лишь %s'), $type, join(', ', array('apc', 'file', 'memcache', 'xcache'))));

 78: 
 79:         $this->type = $type;

148: 
149:         switch ($this->type) {
150:             case 'apc':
151:                 $value = apc_fetch($key);
152:                 if ($value === FALSE) {
153:                     return $default;
154:                 }
155:                 return unserialize($value);
156: 
157:             case 'file':
158:                 if (isset($this->cache[$key])) {
159:                     $expire = $this->cache[$key]['expire'];
160:                     if (!$expire || $expire >= time()) {
161:                         return $this->cache[$key]['value'];
162:                     } elseif ($expire) {
163:                         unset($this->cache[$key]);
164:                         $this->state = 'dirty';
165:                     }
166:                 }

178:     public function save() {
179:         if ($this->type != 'file') {
180:             return;
181:         }
182: 
183:         // Randomly clean the cache out
184:         if (rand(0, 99) == 50) {
185:             $clear_before = time();
186: 
187:             foreach ($this->cache as $key => $value) {
188:                 if ($value['expire'] && $value['expire'] < $clear_before) {
189:                     unset($this->cache[$key]);

192:             }

195:         if ($this->state == 'clean') {
