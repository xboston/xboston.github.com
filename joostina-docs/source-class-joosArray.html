21:     /**
22:      * ArrayTools::get_position()
23:      *
24:      * Определение позиции (порядкового номера) элемента массива по ключу
25:      *
26:      * @param mixed $array
27:      * @param mixed $cur
28:      *
29:      * @return integer Порядковый номер элемента в массиве
30:      */
31:     public static function get_position(array $array, $cur) {
32:         $i = 0;
33:         foreach ($array as $key => $val) {
34:             if ($key == $cur) {
35:                 return $i;
36:             }
37:             $i++;
38:         }
39:         return null;
40:     }
41: 
42:     /**
43:      * Возвращает срез массива по заданным параметрам:
44:      *
45:      * @param array   $array        Исходный массив
46:      * @param mixed   $cur_key      Ключ "текущего" элемента (вокруг которого делаем срез)
47:      * @param integer $prev         Количество элементов, предшествующих текущему
48:      * @param integer $next         Количество элементов, следуемых за текущим
49:      * @param bool    $static_count Флаг, повзоляющий всегда сохранять требуемое количество элементов в возвращаемом массиве
50:      *
51:      * @return array
52:      *
53:      * TODO: добавить возможность не вычислять позицию элемента, если в качестве ключа передается реальный порядковый номер
54:      */
55:     public static function get_next_prev($array, $cur_key, $prev = 3, $next = 5, $static_count = false) {
56:         $position = self::get_position($array, $cur_key);
57:         $offset = $position - $prev > 0 ? $position - $prev : 0;
58:         $length = $prev + $next + 1;
59: 
60:         if ($static_count) {
61:             $count = count($array);
62:             $last_index = $count - 1;
63: 
64:             if (( $last_index - $offset ) < ( $prev + $next )) {
65:                 $offset = $offset + ( $last_index - $offset ) - ( $prev + $next );
66:             }
67:         }
68:         return array_slice($array, $offset, $length, true);
69:     }
