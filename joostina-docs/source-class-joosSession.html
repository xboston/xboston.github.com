 31:         }

 40: 
 41:         $user_ip = joosRequest::user_ip();

 43: 
 44:         return joosCSRF::hash($user_ip . $user_browser);
 45:     }

 48:         self::start();

 57:     public static function session_cookie_name() {
 58:         $syte = str_replace(array('http://', 'https://'), '', JPATH_SITE);
 59:         return md5('site' . $syte);
 60:     }
 61: 
 62:     public static function session_cookie_value($id = null) {
 63:         $type = joosConfig::get2('session', 'type', 3);
 64:         $browser = @$_SERVER['HTTP_USER_AGENT'];
 65: 
 66:         switch ($type) {
 67:             case 2:
 68:                 $value = md5($id . $_SERVER['REMOTE_ADDR']);
 69:                 break;
 70: 
 71:             case 1:
 72:                 // slightly reduced security - 3rd level IP authentication for those behind IP Proxy
 73:                 $remote_addr = explode('.', $_SERVER['REMOTE_ADDR']);
 74:                 $ip = $remote_addr[0] . '.' . $remote_addr[1] . '.' . $remote_addr[2];
 75:                 $value = joosCSRF::hash($id . $ip . $browser);
 76:                 break;
 77: 
 78:             default:
 79:                 $ip = $_SERVER['REMOTE_ADDR'];
 80:                 $value = joosCSRF::hash($id . $ip . $browser);
 81:                 break;
 82:         }
 83: 
 84:         return $value;
 85:     }
