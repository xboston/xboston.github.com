294:     private static function get_file_context($file, $line_number) {
295: 
296:         $context = array();
297:         $i = 0;
298:         foreach (file($file) as $line) {
299:             $i++;
300:             if ($i >= $line_number - 3 && $i <= $line_number + 3) {
301:                 if ($i == $line_number) {
302:                     $context[] = ' >>   ' . $i . "\t" . $line;
303:                 } else {
304:                     $context[] = "\t" . $i . "\t" . $line;
305:                 }
306:             }
307:             if ($i > $line_number + 3) {
308:                 break;
309:             }
310:         }
311: 
312:         return "\n" . implode("", $context);
313:     }
