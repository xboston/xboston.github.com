313:     protected static $counters = array(
314:         'string' => 'joosString::strlen',
315:         'array' => 'count',
316:         'list' => 'count',
317:         'alnum' => 'joosString::strlen',
318:         'alpha' => 'joosString::strlen',
319:         'digit' => 'joosString::strlen',

321:         'space' => 'joosString::strlen',
322:         'upper' => 'joosString::strlen',
323:     );

325:     public static function valid($value, $expected, $message = false) {
326:         foreach (explode('|', $expected) as $item) {
327:             list($type) = $item = explode(':', $item, 2);
328:             // проверяем тип переменной
329:             if (isset(static::$validators[$type])) {
330:                 if (!call_user_func(static::$validators[$type], $value)) {
331:                     continue;
332:                 }
333:             } elseif ($type === 'number') {
334:                 if (!is_int($value) && !is_float($value)) {
335:                     continue;
336:                 }
337:                 // проверка через произвольные регулярки
338:             } elseif ($type === 'pattern') {
339:                 if (preg_match('|^' . (isset($item[1]) ? $item[1] : '') . '$|', $value)) {
340:                     return TRUE;
341:                 }
342:                 continue;
343:             } elseif (!$value instanceof $type) {
344:                 continue;
345:             }
346: 
347:             // проверяем вхождение в разрешённый диапазон
348:             if (isset($item[1])) {
349:                 if (isset(static::$counters[$type])) {
350:                     $value = call_user_func(static::$counters[$type], $value);
351:                 }
352:                 $range = explode('..', $item[1]);
353:                 if (!isset($range[1])) {
354:                     $range[1] = $range[0];
355:                 }
356:                 if (($range[0] !== '' && $value < $range[0]) || ($range[1] !== '' && $value > $range[1])) {
357:                     //если передана строка для форматирования предупреждения - сформируем её
358:                     $message = $message == false ? false : strtr($message, array(':min' => $range[0], ':max' => $range[1]));
359:                     continue;
360:                 }
361:             }
362:             return TRUE;
363:         }
364:         return $message;
365:     }
