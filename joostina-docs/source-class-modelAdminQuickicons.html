118: 
119:         $path        = JPATH_BASE . DS . 'app' . DS . 'components' . DS;
120: 
121:         $controllers = array ();
122:         foreach ( glob( $path . '*' , GLOB_ONLYDIR ) as $controller ) {
123:             $controller               = str_replace( $path , '' , $controller );
124:             $controllers[$controller] = $controller;
