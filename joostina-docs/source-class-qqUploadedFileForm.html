 98:     public static function toBytes($str) {
 99:         $val = trim($str);
100:         $last = strtolower($str[strlen($str) - 1]);
101:         switch ($last) {
102:             case 'g': $val *= 1024;

112:     function handleUpload($uploadDirectory, $replaceOldFile = FALSE, $new_file_name = false, $only_check = false) {
113:         if (!is_writable($uploadDirectory)) {
114:             return array('error' => "Server error. Upload directory isn't writable.");
115:         }
116: 
117:         if (!$this->file) {
118:             return array('error' => 'No files were uploaded.');

122: 
123:         if ($size == 0) {
124:             return array('error' => 'File is empty');
125:         }

128:             return array('error' => 'File is too large');
129:         }

133:         $ext = $pathinfo['extension'];

135:         if ($this->allowedExtensions && !in_array(strtolower($ext), $this->allowedExtensions)) {

137:             return array('error' => 'File has an invalid extension, it should be one of ' . $these . '.');

147:                 $filename .= rand(10, 99);
